SET NOCOUNT ON;

BEGIN TRY
    BEGIN TRAN;

    -- เติมเฉพาะคู่ที่ยังไม่มีใน ProductStocks
    INSERT INTO miniapp.dbo.ProductStocks
        (ProductId, LocationId, QtyOnHand, QtyReserved, QtyDamaged, MinLevel, MaxLevel, ReorderPoint, Cost)
    SELECT
        p.Id         AS ProductId,
        l.Id         AS LocationId,
        0            AS QtyOnHand,
        0            AS QtyReserved,
        0            AS QtyDamaged,
        NULL         AS MinLevel,
        NULL         AS MaxLevel,
        NULL         AS ReorderPoint,
        NULL         AS Cost
    FROM miniapp.dbo.Products  AS p
    CROSS JOIN miniapp.dbo.Locations AS l
    LEFT JOIN miniapp.dbo.ProductStocks AS ps
           ON ps.ProductId = p.Id AND ps.LocationId = l.Id
    WHERE ps.Id IS NULL;  -- แถวที่ยังไม่มีเท่านั้น

    PRINT CONCAT('Inserted missing ProductStocks rows: ', @@ROWCOUNT);

    COMMIT TRAN;
END TRY
BEGIN CATCH
    IF @@TRANCOUNT > 0 ROLLBACK TRAN;
    THROW;
END CATCH;
